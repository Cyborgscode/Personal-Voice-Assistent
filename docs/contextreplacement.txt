Basic-Syntax:

contextreplacements:"INTERNALCOMMAND","replacetext","replacement"

contextreplacements:"AUTHORIZE","neuen","neun"
contextreplacements:"STARTAPP","öffner","öffne"
contextreplacements:"STARTAPP","starter","start"

contextreplacements:"ADDTITLE","föge","füge"
